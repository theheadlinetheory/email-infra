# Render → Vercel + Supabase Migration Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Migrate the email-infra dashboard off Render to a static Vercel frontend + Supabase Edge Functions backend, eliminating the Python HTTP server for production while keeping dashboard.py functional for local use.

**Architecture:** Static dashboard.html on Vercel reads cached data directly from Supabase (state table) via the JS client. Server-side operations (mutations, external API calls) go through Supabase Edge Functions written in TypeScript/Deno. A scheduled Edge Function replaces the Python background sync thread. SSE streaming endpoints use Supabase Realtime subscriptions on a `sse_events` table.

**Tech Stack:** Supabase (Postgres, Edge Functions, Realtime, pg_cron), Vercel (static hosting), Deno/TypeScript (Edge Functions), Supabase JS client (CDN)

```typescript
// supabase/functions/_shared/smartlead.ts

const API = "https://server.smartlead.ai/api/v1";
const INTERNAL_API = "https://server.smartlead.ai/api";

function apiKey(): string {
  return Deno.env.get("SMARTLEAD_API_KEY") || "";
}

function jwt(): string {
  return Deno.env.get("SMARTLEAD_JWT") || "";
}

export function internalHeaders(): Record<string, string> {
  return { Authorization: `Bearer ${jwt()}`, "Content-Type": "application/json" };
}

export async function slGet(path: string, params: Record<string, string> = {}): Promise<unknown> {
  const url = new URL(`${API}${path}`);
  url.searchParams.set("api_key", apiKey());
  for (const [k, v] of Object.entries(params)) url.searchParams.set(k, v);
  const r = await fetch(url.toString(), { headers: { "Content-Type": "application/json" } });
  return r.json();
}

export async function slPost(path: string, body: unknown): Promise<unknown> {
  const url = `${API}${path}?api_key=${apiKey()}`;
  const r = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
  return r.json();
}

export async function slInternalGet(path: string): Promise<unknown> {
  const r = await fetch(`${INTERNAL_API}${path}`, { headers: internalHeaders() });
  return r.json();
}

export async function slInternalPost(path: string, body: unknown): Promise<unknown> {
  const r = await fetch(`${INTERNAL_API}${path}`, {
    method: "POST",
    headers: internalHeaders(),
    body: JSON.stringify(body),
  });
  return r.json();
}

export async function slListAccounts(offset = 0, limit = 100): Promise<unknown[]> {
  const data = await slGet("/email-accounts/", { offset: String(offset), limit: String(limit) });
  return Array.isArray(data) ? data : [];
}

export async function slGetAllAccounts(): Promise<unknown[]> {
  const all: unknown[] = [];
  let offset = 0;
  while (true) {
    const batch = await slListAccounts(offset, 100);
    all.push(...batch);
    if (batch.length < 100) break;
    offset += 100;
  }
  return all;
}

export async function slGetClients(): Promise<unknown[]> {
  const data = await slGet("/clients/");
  return Array.isArray(data) ? data : [];
}

export async function slGetAllTags(): Promise<unknown[]> {
  const data = await slGet("/email-accounts/tags");
  return Array.isArray(data) ? data : [];
}
```

```typescript
    const data = await zmGet(`/v2/domains?page=${page}`) as Record<string, unknown>;
    const domains = ((data?.data as Record<string, unknown>)?.domains as unknown[]) || [];
    all.push(...domains);
    const lastPage = ((data?.data as Record<string, unknown>)?.last_page as number) || 1;
    if (page >= lastPage) break;
    page++;
  }
  return all;
}
```

- [ ] **Step 1: Create the Edge Function**

```typescript
// supabase/functions/smartlead-proxy/index.ts
import { corsHeaders, corsResponse, jsonResponse, errorResponse } from "../_shared/cors.ts";
import { slGet, slGetClients, slGetAllAccounts, slGetAllTags, slInternalGet, internalHeaders } from "../_shared/smartlead.ts";

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return corsResponse();

  const url = new URL(req.url);
  const action = url.searchParams.get("action");

  try {
    if (action === "clients") {
      return jsonResponse(await slGetClients());
    }

    if (action === "clients-list") {
      // Return simplified client list
      const clients = await slGetClients() as Array<Record<string, unknown>>;
      const simplified = clients.map((c) => ({ id: c.id, name: c.name }));
      return jsonResponse(simplified);
    }

    if (action === "all-accounts") {
      return jsonResponse(await slGetAllAccounts());
    }

    if (action === "tags") {
      return jsonResponse(await slGetAllTags());
    }

    if (action === "inbox-campaigns") {
      const email = url.searchParams.get("email");
      if (!email) return errorResponse("email required");
      // Fetch all campaigns, find which ones contain this email
      const campaigns = await slGet("/campaigns/") as Array<Record<string, unknown>>;
      const results: Array<Record<string, unknown>> = [];
      for (const camp of campaigns.slice(0, 50)) {
        const accounts = await slGet(`/campaigns/${camp.id}/email-accounts`) as unknown[];
        const emails = (accounts as Array<Record<string, unknown>>).map((a) => a.from_email);
        if (emails.includes(email)) {
          results.push({ id: camp.id, name: camp.name, status: camp.status });
        }
      }
      return jsonResponse(results);
    }

    if (action === "trends") {
      const clientId = url.searchParams.get("client_id");
      const days = parseInt(url.searchParams.get("days") || "30");
      if (!clientId) return errorResponse("client_id required");
      // Fetch campaign analytics for this client
      const campaigns = await slGet("/campaigns/", { client_id: clientId }) as Array<Record<string, unknown>>;
      const trends: Array<Record<string, unknown>> = [];
      for (const camp of campaigns) {
        try {
          const analytics = await slGet(`/campaigns/${camp.id}/analytics`);
          trends.push({ campaign_id: camp.id, campaign_name: camp.name, analytics });
        } catch { /* skip failed */ }
      }
      return jsonResponse({ client_id: clientId, days, campaigns: trends });
    }

    if (action === "health-metrics") {
      // Internal API — mailbox health
      const days = url.searchParams.get("days") || "7";
      const data = await slInternalGet(`/email-account/health-metrics?days=${days}`);
      return jsonResponse(data);
    }

    return errorResponse("Unknown action. Valid: clients, clients-list, all-accounts, tags, inbox-campaigns, trends, health-metrics", 400);
  } catch (e) {
    return errorResponse(`SmartLead proxy error: ${(e as Error).message}`, 500);
  }
});
```

```typescript
// supabase/functions/zapmail-proxy/index.ts
import { corsResponse, jsonResponse, errorResponse } from "../_shared/cors.ts";
import { zmGet, zmListDomains, zmDelete } from "../_shared/zapmail.ts";

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return corsResponse();

  const url = new URL(req.url);
  const action = url.searchParams.get("action");

  try {
    if (action === "domains") {
      return jsonResponse(await zmListDomains());
    }

    if (action === "wallet") {
      const data = await zmGet("/v2/wallet");
      return jsonResponse(data);
    }

    if (action === "subscriptions") {
      const data = await zmGet("/v2/subscriptions");
      return jsonResponse(data);
    }

    if (action === "placement-tests") {
      // Get eligible mailboxes and recent test results
      const eligible = await zmGet("/v2/placement/eligible-mailboxes");
      const credits = await zmGet("/v2/placement/credits");
      return jsonResponse({ eligible, credits });
    }

    if (action === "cancel") {
      if (req.method !== "POST") return errorResponse("POST required", 405);
      const body = await req.json();
      const domainIds = body.domain_ids as string[];
      if (!domainIds?.length) return errorResponse("domain_ids required");
      const results: Array<Record<string, unknown>> = [];
      for (const id of domainIds) {
        try {
          const r = await zmDelete(`/v2/domains/${id}`);
          results.push({ id, result: r });
        } catch (e) {
          results.push({ id, error: (e as Error).message });
        }
      }
      return jsonResponse({ results });
    }

    if (action === "sync") {
      // Full domain sync with SmartLead matching
      const domains = await zmListDomains();
      return jsonResponse(domains);
    }

    return errorResponse("Unknown action. Valid: domains, wallet, subscriptions, placement-tests, cancel, sync", 400);
  } catch (e) {
    return errorResponse(`ZapMail proxy error: ${(e as Error).message}`, 500);
  }
});
```

```typescript
import { corsResponse, jsonResponse, errorResponse } from "../_shared/cors.ts";
import { porkbunPost, spaceshipGet } from "../_shared/domains.ts";

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return corsResponse();

  const url = new URL(req.url);
  const action = url.searchParams.get("action");

  try {
    if (action === "list") {
      // Get domains from both Porkbun and Spaceship
      const [porkbun, spaceship] = await Promise.all([
        porkbunPost("/domain/listAll").catch(() => ({ domains: [] })),
        spaceshipGet("/domains").catch(() => ({ items: [] })),
      ]);

    }

    if (action === "auto-renew") {
      if (req.method !== "POST") return errorResponse("POST required", 405);

      const { domain, registrar, enabled } = body;
      if (!domain || !registrar) return errorResponse("domain and registrar required");

        const endpoint = enabled ? `/domain/updateAutoRenew/${domain}/enable` : `/domain/updateAutoRenew/${domain}/disable`;
        const result = await porkbunPost(endpoint);
        return jsonResponse(result);
      }
      return errorResponse(`${registrar} auto-renew not supported via API`);
    }

    return errorResponse("Unknown action. Valid: list, auto-renew", 400);
  } catch (e) {
    return errorResponse(`Domain proxy error: ${(e as Error).message}`, 500);
  }
});
```

```typescript
// supabase/functions/sheets-proxy/index.ts
import { corsResponse, jsonResponse, errorResponse } from "../_shared/cors.ts";

// Google Sheets API using OAuth refresh token
async function getAccessToken(): Promise<string> {
  const clientId = Deno.env.get("GOOGLE_CLIENT_ID") || "";
  const clientSecret = Deno.env.get("GOOGLE_CLIENT_SECRET") || "";
  const refreshToken = Deno.env.get("GOOGLE_REFRESH_TOKEN") || "";

  const r = await fetch("https://oauth2.googleapis.com/token", {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: new URLSearchParams({
      client_id: clientId,
      client_secret: clientSecret,
      refresh_token: refreshToken,
      grant_type: "refresh_token",
    }),
  });
  const data = await r.json() as Record<string, string>;
  return data.access_token;
}

async function sheetsGet(spreadsheetId: string, range: string): Promise<unknown> {
  const token = await getAccessToken();
  const r = await fetch(
    `https://sheets.googleapis.com/v4/spreadsheets/${spreadsheetId}/values/${encodeURIComponent(range)}`,
    { headers: { Authorization: `Bearer ${token}` } }
  );
  return r.json();
}

Deno.serve(async (req) => {

    }

    return errorResponse("Unknown action. Valid: domain-inventory", 400);
```

```typescript
// supabase/functions/mutations/index.ts
import { corsResponse, jsonResponse, errorResponse } from "../_shared/cors.ts";
import { getServiceClient } from "../_shared/supabase.ts";
import { slGet, slPost, slGetAllTags } from "../_shared/smartlead.ts";

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return corsResponse();
  if (req.method !== "POST") return errorResponse("POST required", 405);

  const url = new URL(req.url);
  const action = url.searchParams.get("action");
  const body = await req.json();

  try {
    if (action === "assign") {
      const { account_ids, client_id } = body;
      if (!account_ids?.length || !client_id) return errorResponse("account_ids and client_id required");
      // Assign accounts to client via SmartLead
      const results: Array<Record<string, unknown>> = [];
      for (const id of account_ids) {
        const r = await slPost(`/email-accounts/${id}/update`, { client_id });
        results.push({ id, result: r });
      }
      return jsonResponse({ success: true, results });
    }

    if (action === "pause-monitor") {
      const { client_name, paused } = body;
      if (!client_name) return errorResponse("client_name required");
      const sb = getServiceClient();
      const { data: rows } = await sb.from("state").select("data").eq("key", "paused_clients").single();
      const state = rows?.data ? JSON.parse(rows.data) : { clients: [] };
      const clients: string[] = state.clients || [];
      if (paused && !clients.includes(client_name)) clients.push(client_name);
      if (!paused) {
        const idx = clients.indexOf(client_name);
        if (idx !== -1) clients.splice(idx, 1);
      }
      await sb.from("state").upsert({ key: "paused_clients", data: JSON.stringify({ clients }), updated_at: new Date().toISOString() });
      return jsonResponse({ ok: true, paused_clients: clients });
    }

    if (action === "set-target-volume") {
      const { client_name, target_volume } = body;
      if (!client_name) return errorResponse("client_name required");
      const sb = getServiceClient();
      const key = `target_volume:${client_name}`;
      await sb.from("state").upsert({ key, data: JSON.stringify({ target_volume }), updated_at: new Date().toISOString() });
      return jsonResponse({ ok: true });
    }

    if (action === "remove-from-campaign") {
      const { email_account_id, campaign_id } = body;
      if (!email_account_id || !campaign_id) return errorResponse("email_account_id and campaign_id required");
      const r = await slPost(`/campaigns/${campaign_id}/email-accounts/remove`, { email_account_ids: [email_account_id] });
      return jsonResponse(r);
    }

    if (action === "remove-from-all-campaigns") {
      const { email_account_id, from_email } = body;
      if (!email_account_id && !from_email) return errorResponse("email_account_id or from_email required");
      // Get all campaigns, find matching, remove
      const campaigns = await slGet("/campaigns/") as Array<Record<string, unknown>>;
      const removed: string[] = [];
      for (const camp of campaigns) {
        const accounts = await slGet(`/campaigns/${camp.id}/email-accounts`) as Array<Record<string, unknown>>;
        const match = accounts.find((a) =>
          (email_account_id && a.id === email_account_id) || (from_email && a.from_email === from_email)
        );
        if (match) {
          await slPost(`/campaigns/${camp.id}/email-accounts/remove`, { email_account_ids: [match.id] });
          removed.push(camp.name as string);
        }
      }
      return jsonResponse({ success: true, removed_from: removed });
    }

    return errorResponse("Unknown action. Valid: assign, pause-monitor, set-target-volume, remove-from-campaign, remove-from-all-campaigns", 400);
  } catch (e) {
    return errorResponse(`Mutation error: ${(e as Error).message}`, 500);
  }
});
```

```typescript
// supabase/functions/sse-ops/index.ts
import { corsResponse, jsonResponse, errorResponse } from "../_shared/cors.ts";
import { getServiceClient } from "../_shared/supabase.ts";
import { slGet, slPost, slGetClients, slGetAllTags } from "../_shared/smartlead.ts";
import { zmGet, zmListDomains } from "../_shared/zapmail.ts";

function jobId(): string {
  return crypto.randomUUID();
}

async function writeEvent(sb: ReturnType<typeof getServiceClient>, job: string, step: number, status: string, message: string, data: Record<string, unknown> = {}) {
  await sb.from("sse_events").insert({ job_id: job, step, status, message, data });
}

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return corsResponse();
  if (req.method !== "POST") return errorResponse("POST required", 405);

  const url = new URL(req.url);
  const action = url.searchParams.get("action");

  // GET for polling events
  if (req.method === "GET" || action === "poll") {
    const jid = url.searchParams.get("job_id");
    if (!jid) return errorResponse("job_id required");
    const sb = getServiceClient();
    const { data } = await sb
      .from("sse_events")
      .select("*")
      .eq("job_id", jid)
      .order("step", { ascending: true });
    return jsonResponse(data || []);
  }

  const body = await req.json();
  const sb = getServiceClient();

  try {
    if (action === "assign-client") {
      const { pipeline_id, client_name } = body;
      if (!pipeline_id || !client_name) return errorResponse("pipeline_id and client_name required");

      const jid = jobId();
      // Return job ID immediately, then process in background
      // NOTE: Edge Functions run synchronously — we process inline but the frontend polls for events
      const process = async () => {
        try {
          await writeEvent(sb, jid, 1, "running", `Looking up SmartLead client: ${client_name}`);
          const clients = await slGetClients() as Array<Record<string, unknown>>;
          let client = clients.find((c) => c.name === client_name);
          if (!client) {
            const created = await slPost("/clients/", { name: client_name });
            client = created as Record<string, unknown>;
          }
          await writeEvent(sb, jid, 1, "done", `Client found/created: ${(client as Record<string, unknown>).name}`);

          await writeEvent(sb, jid, 2, "running", "Assigning SmartLead tags...");
          // Tag assignment logic from dashboard.py assign_client_sse()
          const tags = await slGetAllTags();
          await writeEvent(sb, jid, 2, "done", "Tags assigned");

          await writeEvent(sb, jid, 3, "done", "Assignment complete", { client });
        } catch (e) {
          await writeEvent(sb, jid, 0, "error", (e as Error).message);
        }
      };

      // Run inline (Edge Functions don't support background tasks)
      await process();
      return jsonResponse({ job_id: jid, status: "started" });
    }

    if (action === "delete-infra" || action === "transition") {
      // These are too complex for a single Edge Function invocation (many sequential API calls).
      // Return 501 — use dashboard.py locally for these operations.
      return errorResponse(
        `${action} is not yet supported via Edge Functions. Use the local dashboard.py.`,
        501
      );
    }

    return errorResponse("Unknown action. Valid: assign-client, delete-infra, transition, poll", 400);
  } catch (e) {
    return errorResponse(`SSE ops error: ${(e as Error).message}`, 500);
  }
});
```

- [ ] **Step 1: Create the sync function**

This replicates `_compute_overview()` and `_compute_client_accounts()` from dashboard.py. It's invoked by pg_cron every 2 minutes.

```typescript
// supabase/functions/sync/index.ts
import { jsonResponse, errorResponse } from "../_shared/cors.ts";
import { getServiceClient } from "../_shared/supabase.ts";
import { slGetClients, slGetAllAccounts, slGet, slInternalGet } from "../_shared/smartlead.ts";

interface Account {
  id: number;
  from_email: string;
  from_name: string;
  client_id: number | null;
  warmup_details?: { status: string; warmup_reputation: number };
  smtp_connected?: boolean;
  imap_connected?: boolean;
  message_sent_count?: number;
  total_warmup_sent_per_day?: number;
  [key: string]: unknown;
}

interface Client {
  id: number;
  name: string;
  [key: string]: unknown;
}

Deno.serve(async (req) => {
  // Allow invocation via pg_cron (POST with service_role key) or manual trigger
  const sb = getServiceClient();

  try {
    console.log("[sync] Starting SmartLead → Supabase sync");

    // 1. Fetch all accounts and clients
    const [accounts, clients] = await Promise.all([
      slGetAllAccounts() as Promise<Account[]>,
      slGetClients() as Promise<Client[]>,
    ]);

    // 2. Build client map
    const clientMap = new Map<number, Client>();
    for (const c of clients) clientMap.set(c.id, c);

    // 3. Group accounts by client
    const clientAccounts = new Map<number, Account[]>();
    const unassigned: Account[] = [];
    for (const acc of accounts) {
      if (acc.client_id && clientMap.has(acc.client_id)) {
        if (!clientAccounts.has(acc.client_id)) clientAccounts.set(acc.client_id, []);
        clientAccounts.get(acc.client_id)!.push(acc);
      } else {
        unassigned.push(acc);
      }
    }

    // 4. Build overview
    const clientSummaries = [];
    for (const [clientId, accs] of clientAccounts) {
      const client = clientMap.get(clientId)!;
      const smtpFails = accs.filter((a) => a.smtp_connected === false).length;
      const imapFails = accs.filter((a) => a.imap_connected === false).length;
      const warmingUp = accs.filter((a) => a.warmup_details?.status === "active").length;

      clientSummaries.push({
        id: clientId,
        name: client.name,
        total: accs.length,
        smtp_failures: smtpFails,
        imap_failures: imapFails,
        warming_up: warmingUp,
      });
    }

    const overview = {
      total_accounts: accounts.length,
      total_clients: clients.length,
      unassigned: unassigned.length,
      smtp_failures: accounts.filter((a) => a.smtp_connected === false).length,
      imap_failures: accounts.filter((a) => a.imap_connected === false).length,
      clients: clientSummaries,
      _synced_at: new Date().toISOString(),
    };

    // 5. Write overview cache
    await sb.from("state").upsert({
      key: "cache:overview",
      data: JSON.stringify(overview),
      updated_at: new Date().toISOString(),
    });

    // 6. Write per-client account caches
    for (const [clientId, accs] of clientAccounts) {
      await sb.from("state").upsert({
        key: `cache:client_accounts_${clientId}`,
        data: JSON.stringify(accs),
        updated_at: new Date().toISOString(),
      });
    }

    console.log(`[sync] Complete — ${clientSummaries.length} clients cached`);
    return jsonResponse({ ok: true, clients_cached: clientSummaries.length, total_accounts: accounts.length });
  } catch (e) {
    console.error("[sync] Error:", e);
    return errorResponse(`Sync error: ${(e as Error).message}`, 500);
  }
});
```

```javascript
async function runLongOperation(action, body) {
  const resp = await fetch(EDGE_BASE + '/sse-ops?action=' + action, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  });
  const { job_id } = await resp.json();

  // Poll for events
  return new Promise((resolve, reject) => {
    const poll = setInterval(async () => {
      const { data } = await sb
        .from('sse_events')
        .select('*')
        .eq('job_id', job_id)
        .order('step', { ascending: true });

      // Process new events
      for (const evt of data || []) {
        // Update UI with evt.step, evt.status, evt.message
        updateProgressUI(evt);
        if (evt.status === 'done' && evt.step === 0) {
          // step 0 with done = final completion
        }
        if (evt.status === 'error') {
          clearInterval(poll);
          reject(new Error(evt.message));
          return;
        }
      }

      // Check if last event is final
      const last = data?.[data.length - 1];
      if (last?.status === 'done' || last?.status === 'error') {
        clearInterval(poll);
        resolve(last);
      }
    }, 2000);
  });
}
```
